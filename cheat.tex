\subsection{Electricity \& Magnetism}
Maxwell's Equations \& Simple E and M
\beqs
  \curl{B} = \mu_0 \vec{J} + \frac{1}{c^2}\D{t}{\vec{E}}
  \qqquad
  \int \vec{B} \cdot \mathrm{d} \ell = \mu_0 \int \vec{J} \cdot \mathrm{d} s
\eeqs
\beqs
  \vec{J} = \frac{\vec{B} \cross \grad \vec{P}}{B^2}
  \qqqquad
  \curl{E} = -\D{t}{\vec{B}}
\eeqs
\beqs
  V = I R
  \qqqquad
  P = I V = I^2 R = \frac{V^2}{R}
\eeqs
Forces
\beqs
  \vec{F} = q \vec{v} \cross \vec{B} = \vec{J} \cross \vec{B} = \grad P
  \quad
  v_{drift} = \frac{\vec{F} \cross \vec{B}}{q B^2}
  \quad
  \vec{F} = q \vec{E}
\eeqs
Waves
\beqs
  \omega = 2 \pi f
  \qqqquad
  c = f \lambda
  \qqqquad
  E_{photon} = h f = \frac{h c}{\lambda}
\eeqs

\subsection{Magnetic Confinement Fusion}
Toroid geometry
\beqs
  R_0 \equiv \mbox{Major radius}
  \quad
  2 a \equiv \mbox{D width}
  \quad
  2 b \equiv \mbox{D height}
\eeqs
\beqs
  A \equiv \frac{R_0}{a}
  \quad
  \mathcal{S}_{ellipse} = \paren{ \pi a b } = \frac{\pi R^2_0 \epsilon}{A^2}
\eeqs
\beqs
  \epsilon \equiv \frac{b}{a}
  \quad
  \mathcal{V}_{toroid} = \paren{ 2 \pi \bar{R} \mathcal{S}_{ellipse} } = \frac{2 \pi^2 R^3_0 \epsilon}{A^2}
\eeqs
Safety Factor \& Plasma Beta
\beqs
  q \paren{r} \equiv \frac{r}{R} \frac{B_\phi}{B_\theta}
  \qquad
  \beta_{pol} \equiv \frac{2 \mu_0 P_{kinetic}}{B_{pol}}
  \qquad
  \beta_{tor} \equiv \frac{\beta_{pol}}{q^2 A^2}
\eeqs
Bennett Z-Pinch (straight B-fields)
\beqs
  p_{kin} + \frac{B^2}{2 \mu_0} = const
  \qquad
  \beta \equiv \frac{p_{kin}}{p_{mag}}
  \qquad
  0 \leq \beta \leq 1
\eeqs

\subsection{Particles}
Maxwell-Boltzmann distributions
\beqs
  \left< v_\alpha \right> = \sqrt{\frac{8}{\pi}\frac{T_\alpha}{m_\alpha}}
  \qquad
  v_{\alpha rms} = \sqrt{\frac{3 T_\alpha}{m_\alpha}}
  \qquad
  \paren{ \frac{v_\alpha}{c} }^2 = \frac{T_\alpha}{m_\alpha c^2}
\eeqs
Kinetic pressure
\beqs
  p = n_e T_e + \sum n_z T_z
  \rightarrow 2 n T (n_e = n_i \& T_e = T_i)
  \quad
  n_e = \sum n_z Z
\eeqs

\subsection{Fusion Power}
Fusion reaction rate
\beqs
  RR_{ab} = \frac{n_a n_b}{1 + \delta_{ab}} \left< \sigma v \right>

\eeqs
Plant Efficiency

\subsection{Confinement}
\beqs n \tau = \frac{12 T}{ \left< \sigma v \right> (f_\alpha (1 - \chi_R) + \frac{1}{Q}) E_{fusion}} \eeqs
\beqs \chi_R = \frac{P_{rad}}{P_\alpha} = 0.055 \sqrt{\frac{T}{15 \units{keV}}} Z_{eff} \eeqs
\beqs
  f_{\alpha} = \frac{P_\alpha}{P_{fusion}} = \frac{1}{5}

\eeqs

\subsection{Transport Coefficients}
Timesteps \& Coulomb logarithms
\beqs
  n \tau_e = 3.44 \E{5} \frac{T^{3/2}_{e,eV}}{\log{\Lambda_e}} \units{\frac{s}{cm^3}}
  \qquad
  \log{\Lambda_e} = 17.5
\eeqs
\beqs
  n \tau_i = 2.09 \E{7} \frac{T^{3/2}_{i,eV}}{\log{\Lambda_i}} \frac{m}{m_p}^{1/2} \units{\frac{s}{cm^3}}
  \qquad
  \log{\Lambda_i} = 15
\eeqs
Electrical Conductivities \& Inter-species heating
\beqs
  \sigma_\perp = n e^2 \frac{\tau_e}{m_{ei}}
  \quad
  \sigma_\parallel = 1.96 \sigma_\perp
  \quad
  \dot{q}^{\prime\prime\prime}_{e\rightarrow i} = 3 \paren{\frac{m_e}{m_i}} \frac{\paren{T_e - T_i}n_i}{\tau_e}
\eeqs
Dimensionless collisionality \& Neoclassical heating
\beqs
  \nu^{*}_i = \frac{q R}{v^{2D}_{th,i} \tau_i}
  \qquad
  \dot{q}^{\prime\prime}_{neo} = \dot{q}^{\prime\prime}_{class} \times X
\eeqs
\beqs
  X = \left(
    \begin{array}{lr}
      2 q^2 A^{3/2} & : \nu^{*} < A^{-3/2} \mbox{ (banana)}\\
      1 & : A^{-3/2} < \nu^{*} < 1.0 \mbox{ (plateau)}\\
      2 q^2 & : 1.0 < \nu^{*} \mbox{ (Pfirsch-Schl\"uter)}
    \end{array}
  \right.
\eeqs

\subsection{Thermal Conduction}
\beqs
  \hat{\kappa}^e = n \tau_e \paren{\frac{T_e}{m_e}}
  \quad
  \kappa^e_\parallel = 3.2 \hat{\kappa}^e
  \quad
  \kappa^e_\perp = \frac{4.7\hat{\kappa}^e}{\paren{\omega_{ce} \tau_e}^2}
  \quad
  \kappa^e_\wedge = \frac{2.5\hat{\kappa}^e}{\paren{\omega_{ce} \tau_e}}
\eeqs
\beqs
  \hat{\kappa}^i = n \tau_i \paren{\frac{T_i}{m_i}}
  \quad
  \kappa^i_\parallel = 3.9 \hat{\kappa}^i
  \quad
  \kappa^i_\perp = \frac{2.0\hat{\kappa}^i}{\paren{\omega_{ci} \tau_i}^2}
  \quad
  \kappa^i_\wedge = \frac{2.5\hat{\kappa}^i}{\paren{\omega_{ci} \tau_i}}
\eeqs
\beqs
  \frac{\kappa^e_\parallel}{\kappa^i_\parallel} \sim 60
  \quad
  \frac{\kappa^e_\perp}{\kappa^i_\perp} \sim \frac{1}{60}
  \rightarrow \frac{4.7 m_e \tau_e}{2 m_i \tau_i} (n_e = n_i \& T_e = T_i)
\eeqs
\beqs
  \dot{q}^{\prime\prime} \equiv - \kappa \grad{T}
  \qquad
  \dot{q}^e_{th} = \dot{q}^e_\parallel + \dot{q}^e_\perp + \dot{q}^e_\wedge
  \qquad
  \dot{q}^i_{th} = \dot{q}^i_\parallel + \dot{q}^i_\perp - \dot{q}^i_\wedge
\eeqs

\subsection{Plasma Parameters}
Gyrofrequency \& thermal velocity \& Debye length
\beqs
  \omega_{c \alpha} = \frac{q B}{m_\alpha}
  \qquad
  v^{XD}_{th} = \sqrt{\frac{X k T}{2 m}}
  \qquad
  \lambda_D = \frac{v^{1D}_{th}}{\omega_{pe}} =\sqrt{\frac{\epsilon_0 k T}{n_\alpha e^2}}
\eeqs
Gyroradius \& Alfven speed
\beqs
  \rho_e = \frac{v^{2D}_{th}}{\omega_{ce}}
  \qquad
  v_A = \frac{B}{\sqrt{\mu_0 \rho}}
\eeqs
Plasma frequency \& plasma dispersion \& Phase velocity
\beqs
  \omega^2_{pe} = \frac{n_e e^2}{\epsilon_0 m_e}
  \qquad
  \omega^2 = \omega^2_{pe} k^2 c^2
  \qquad
  v_{\phi} = \frac{\omega}{k}
\eeqs
